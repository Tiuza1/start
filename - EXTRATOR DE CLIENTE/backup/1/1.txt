import os, subprocess, re, time
import xml.etree.cElementTree as ET
import csv

# --- CONFIGURAÇÕES DE IDs ---
ID_NOME = 'com.jtexpress.braout:id/tv_wait_name'
ID_BOTAO_TEL = 'com.jtexpress.braout:id/iv_wait_phone'
ID_PEDIDO = 'com.jtexpress.braout:id/tv_bill_code'

def parse_bounds(bounds_str):
    nums = re.findall(r'\d+', bounds_str)
    x1, y1, x2, y2 = map(int, nums)
    return (x1 + x2) // 2, (y1 + y2) // 2

def extrair_numero_do_discador():
    time.sleep(1.2) # Tempo para o discador carregar
    subprocess.run("adb shell uiautomator dump /sdcard/dialer.xml", shell=True, capture_output=True)
    subprocess.run("adb pull /sdcard/dialer.xml .", shell=True, capture_output=True)
    
    try:
        tree = ET.parse("dialer.xml")
        for n in tree.getroot().iter('node'):
            text = n.get('text')
            if text:
                limpo = re.sub(r'\D', '', text)
                if len(limpo) >= 8: return limpo
    except: pass
    return "NÃO ENCONTRADO"

def rolar_tela():
    """Faz um movimento de arrastar para cima (scroll down)"""
    # Swipe de baixo para cima: x_inicial y_inicial x_final y_final duração
    # Ajustado para rolar aproximadamente a altura de 3 ou 4 cards
    subprocess.run("adb shell input swipe 500 1500 500 600 500", shell=True)
    time.sleep(1.5) # Espera a animação do scroll terminar

def rodar_extrator_v2():
    arquivo_csv = "clientes_do_dia.csv"
    processados = set()
    tentativas_vazio = 0

    if os.path.exists(arquivo_csv):
        with open(arquivo_csv, 'r', encoding='utf-8') as f:
            reader = csv.reader(f)
            next(reader, None)
            for row in reader: processados.add(row[1])

    print("="*40)
    print("   EXTRATOR 100% AUTOMÁTICO - J&T")
    print("   Pressione Ctrl+C para parar")
    print("="*40)

    try:
        while True:
            subprocess.run("adb shell uiautomator dump /sdcard/v.xml", shell=True, capture_output=True)
            subprocess.run("adb pull /sdcard/v.xml .", shell=True, capture_output=True)
            
            try:
                tree = ET.parse("v.xml")
                nodes = list(tree.getroot().iter('node'))
                
                nomes = [n for n in nodes if n.get('resource-id') == ID_NOME]
                pedidos = [n for n in nodes if n.get('resource-id') == ID_PEDIDO]
                botoes_tel = [n for n in nodes if n.get('resource-id') == ID_BOTAO_TEL]

                novos_nesta_tela = 0

                for i in range(len(nomes)):
                    id_pacote = pedidos[i].get('text')
                    
                    if id_pacote in processados:
                        continue

                    nome_cliente = nomes[i].get('text')
                    x, y = parse_bounds(botoes_tel[i].get('bounds'))

                    print(f"[*] Extraindo: {nome_cliente}")
                    
                    # 1. Clica no telefone
                    subprocess.run(f"adb shell input tap {x} {y}", shell=True)
                    
                    # 2. Pega número
                    telefone = extrair_numero_do_discador()
                    
                    # 3. VOLTAR 2X (Para sair do discador e teclado se houver)
                    subprocess.run("adb shell input keyevent 4", shell=True)
                    time.sleep(0.3)
                    subprocess.run("adb shell input keyevent 4", shell=True)
                    time.sleep(0.5)

                    # 4. Salva
                    with open(arquivo_csv, 'a', newline='', encoding='utf-8') as f:
                        writer = csv.writer(f)
                        if os.path.getsize(arquivo_csv) == 0:
                            writer.writerow(["Nome", "Pacote", "Telefone"])
                        writer.writerow([nome_cliente, id_pacote, telefone])

                    processados.add(id_pacote)
                    novos_nesta_tela += 1
                    tentativas_vazio = 0 # Resetamos o contador de fim de lista

                if novos_nesta_tela == 0:
                    tentativas_vazio += 1
                    print(f"[i] Nenhum novo pedido nesta tela. Rolando... ({tentativas_vazio}/3)")
                    rolar_tela()
                else:
                    print(f"    --> {novos_nesta_tela} novos coletados. Rolando...")
                    rolar_tela()

                # Se rolou 3 vezes e não achou nada novo, provavelmente chegou ao fim
                if tentativas_vazio >= 3:
                    print("\n[✓] Fim da lista alcançado ou não há mais dados.")
                    break

            except Exception as e:
                print(f"Erro no processamento: {e}")
                time.sleep(2)

    except KeyboardInterrupt:
        print("\n[!] Automação interrompida pelo usuário.")

if __name__ == "__main__":
    rodar_extrator_v2()
