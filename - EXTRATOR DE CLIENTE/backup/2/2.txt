import os, subprocess, re, time
import xml.etree.cElementTree as ET
import csv
from datetime import datetime

# --- CONFIGURAÇÕES DE IDs ---
ID_NOME = 'com.jtexpress.braout:id/tv_wait_name'
ID_BOTAO_TEL = 'com.jtexpress.braout:id/iv_wait_phone'
ID_PEDIDO = 'com.jtexpress.braout:id/tv_bill_code'

# Agora rastreamos o par (nome + número)
ultimo_nome_salvo = ""
ultimo_numero_salvo = ""

def parse_bounds(bounds_str):
    nums = re.findall(r'\d+', bounds_str)
    x1, y1, x2, y2 = map(int, nums)
    return (x1 + x2) // 2, (y1 + y2) // 2

def extrair_do_discador_puro():
    if os.path.exists("dialer.xml"): os.remove("dialer.xml")
    subprocess.run("adb shell uiautomator dump /sdcard/dialer.xml", shell=True, capture_output=True)
    subprocess.run("adb pull /sdcard/dialer.xml .", shell=True, capture_output=True)
    try:
        tree = ET.parse("dialer.xml")
        for n in tree.getroot().iter('node'):
            text = n.get('text')
            if text:
                limpo = re.sub(r'\D', '', text)
                if len(limpo) >= 8: return limpo
    except: pass
    return ""

def voltar_ao_app():
    subprocess.run("adb shell input keyevent 4", shell=True)
    time.sleep(0.5)
    subprocess.run("adb shell input keyevent 4", shell=True)
    time.sleep(1.2)

def rolar_tela_lento():
    """Scroll otimizado: Rápido o suficiente para não irritar, lento o suficiente para não pular"""
    subprocess.run("adb shell input swipe 500 1100 500 600 800", shell=True)
    time.sleep(1.0)

def capturar_com_tentativas(x, y, nome_cliente):
    """Clica e valida se o número pertence ao cliente atual"""
    global ultimo_numero_salvo, ultimo_nome_salvo
    
    for tentativa in range(3):
        subprocess.run(f"adb shell input tap {x} {y}", shell=True)
        time.sleep(1.8)
        
        numero_atual = extrair_do_discador_puro()
        
        # --- LÓGICA DE VALIDAÇÃO INTELIGENTE ---
        # 1. Se o número veio vazio, tenta de novo.
        if numero_atual == "":
            print(f"      [!] Aviso: Discador vazio para {nome_cliente}. Tentando {tentativa+1}/3...")
            voltar_ao_app()
            continue

        # 2. Se o número é igual ao anterior MAS o nome mudou, é ERRO de sincronismo.
        if numero_atual == ultimo_numero_salvo and nome_cliente != ultimo_nome_salvo:
            print(f"      [!] BUG DETECTADO: Número de {nome_cliente} está repetindo o anterior. Tentando {tentativa+1}/3...")
            voltar_ao_app()
            time.sleep(1)
            continue
            
        # 3. Se chegou aqui, ou o número é novo, ou é o mesmo cliente (pacote repetido).
        ultimo_numero_salvo = numero_atual
        ultimo_nome_salvo = nome_cliente
        voltar_ao_app()
        return numero_atual
            
    return numero_atual if numero_atual != "" else "ERRO_NA_LEITURA"

def rodar_extrator_v2_4():
    # Gera a data atual no formato 10-04-2026
    data_formatada = datetime.now().strftime("%d-%m-%Y")
    arquivo_csv = f"clientes_do_dia ({data_formatada}).csv"
    
    processados = set()

    if os.path.exists(arquivo_csv):
        with open(arquivo_csv, 'r', encoding='utf-8') as f:
            reader = csv.reader(f)
            next(reader, None)
            for row in reader:
                if len(row) > 1: processados.add(row[1])

    print("="*40)
    print("   EXTRATOR INTELIGENTE V 2.4")
    print("   MODO: MULTI-PACOTES + ANTI-BUG")
    print("="*40)

    try:
        while True:
            if os.path.exists("v.xml"): os.remove("v.xml")
            subprocess.run("adb shell uiautomator dump /sdcard/v.xml", shell=True, capture_output=True)
            subprocess.run("adb pull /sdcard/v.xml .", shell=True, capture_output=True)
            
            try:
                tree = ET.parse("v.xml")
                nodes = list(tree.getroot().iter('node'))
                
                nomes = [n for n in nodes if n.get('resource-id') == ID_NOME]
                pedidos = [n for n in nodes if n.get('resource-id') == ID_PEDIDO]
                botoes_tel = [n for n in nodes if n.get('resource-id') == ID_BOTAO_TEL]

                total_visivel = min(len(nomes), len(pedidos), len(botoes_tel))
                novos_nesta_tela = 0

                for i in range(total_visivel):
                    id_pacote = pedidos[i].get('text')
                    if id_pacote in processados: continue

                    nome_cliente = nomes[i].get('text')
                    x, y = parse_bounds(botoes_tel[i].get('bounds'))

                    print(f"[*] Processando: {nome_cliente} (Pacote: {id_pacote})")
                    
                    telefone = capturar_com_tentativas(x, y, nome_cliente)

                    with open(arquivo_csv, 'a', newline='', encoding='utf-8') as f:
                        writer = csv.writer(f)
                        if os.path.getsize(arquivo_csv) == 0:
                            writer.writerow(["Nome", "Pacote", "Telefone"])
                        writer.writerow([nome_cliente, id_pacote, telefone])

                    processados.add(id_pacote)
                    novos_nesta_tela += 1

                if novos_nesta_tela == 0:
                    rolar_tela_lento()
                else:
                    rolar_tela_lento()

            except Exception as e:
                print(f"Erro: {e}")
                time.sleep(2)
                rolar_tela_lento()

    except KeyboardInterrupt:
        print("\n[!] Finalizado pelo usuário.")

if __name__ == "__main__":
    rodar_extrator_v2_4()
